```typescript
// src/lib/db.ts
import Database from 'better-sqlite3';
import path from 'path';
import crypto from 'crypto';
import type { Blessing, BlessingCreateInput } from './types';

const DB_PATH = path.join(process.cwd(), 'blessings.db');
const db = new Database(DB_PATH);
db.pragma('journal_mode = WAL');

db.exec(`
  CREATE TABLE IF NOT EXISTS blessings (
    id          TEXT PRIMARY KEY,
    name        TEXT NOT NULL,
    birthday    TEXT,
    photos      TEXT DEFAULT '[]',
    voice       TEXT,
    video_url   TEXT,
    letter      TEXT,
    easter_egg  TEXT,
    sender      TEXT,
    created_at  TEXT DEFAULT (datetime('now'))
  )
`);

function generateId(): string {
  return crypto.randomBytes(4).toString('hex');
}

export function createBlessing(input: BlessingCreateInput): string {
  let id: string;
  do {
    id = generateId();
  } while (db.prepare('SELECT 1 FROM blessings WHERE id = ?').get(id));

  const stmt = db.prepare(`
    INSERT INTO blessings (id, name, birthday, photos, voice, video_url, letter, easter_egg, sender)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
  `);
  stmt.run(
    id,
    input.name,
    input.birthday || null,
    JSON.stringify(input.photos),
    input.voice || null,
    input.videoUrl || null,
    input.letter || null,
    input.easterEgg ? JSON.stringify(input.easterEgg) : null,
    input.sender || null,
  );
  return id;
}

export function getBlessing(id: string): Blessing | null {
  const row = db.prepare('SELECT * FROM blessings WHERE id = ?').get(id) as any;
  if (!row) return null;
  return {
    id: row.id,
    name: row.name,
    birthday: row.birthday,
    photos: JSON.parse(row.photos || '[]'),
    voice: row.voice,
    videoUrl: row.video_url,
    letter: row.letter,
    easterEgg: row.easter_egg ? JSON.parse(row.easter_egg) : null,
    sender: row.sender,
    createdAt: row.created_at,
  };
}
```

```typescript
// src/app/api/create/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { writeFile, mkdir } from 'fs/promises';
import path from 'path';
import QRCode from 'qrcode';
import { createBlessing } from '@/lib/db';
import type { BlessingCreateInput } from '@/lib/types';

export async function POST(request: NextRequest) {
  try {
    const formData = await request.formData();

    const name = formData.get('name')?.toString().trim();
    if (!name) {
      return NextResponse.json({ success: false, error: '请输入寿星的姓名' }, { status: 400 });
    }

    const uploadDir = path.join(process.cwd(), 'public', 'uploads');
    await mkdir(uploadDir, { recursive: true });

    const uploadedPhotos: string[] = [];

    // Process multiple photos
    const photoFiles = formData.getAll('photos');
    for (const file of photoFiles) {
      if (file instanceof File) {
        const buffer = Buffer.from(await file.arrayBuffer());
        const filename = `photo-${Date.now()}-${Math.random().toString(36).slice(2, 8)}${path.extname(file.name) || '.jpg'}`;
        const filepath = path.join(uploadDir, filename);
        await writeFile(filepath, buffer);
        uploadedPhotos.push(`/uploads/${filename}`);
      }
    }

    // Process voice
    let voicePath: string | undefined;
    const voiceFile = formData.get('voice');
    if (voiceFile instanceof File && voiceFile.size > 0) {
      const buffer = Buffer.from(await voiceFile.arrayBuffer());
      const filename = `voice-${Date.now()}.webm`;
      await writeFile(path.join(uploadDir, filename), buffer);
      voicePath = `/uploads/${filename}`;
    }

    // Process video
    let videoPath: string | undefined;
    const videoFile = formData.get('video');
    if (videoFile instanceof File && videoFile.size > 0) {
      const buffer = Buffer.from(await videoFile.arrayBuffer());
      const ext = path.extname(videoFile.name) || '.mp4';
      const filename = `video-${Date.now()}${ext}`;
      await writeFile(path.join(uploadDir, filename), buffer);
      videoPath = `/uploads/${filename}`;
    }

    // Parse easter egg
    let easterEgg = undefined;
    const easterEggStr = formData.get('easterEgg')?.toString();
    if (easterEggStr) {
      try {
        easterEgg = JSON.parse(easterEggStr);
      } catch { /* ignore parse errors */ }
    }

    const input: BlessingCreateInput = {
      name,
      birthday: formData.get('birthday')?.toString() || undefined,
      photos: uploadedPhotos,
      voice: voicePath,
      videoUrl: videoPath,
      letter: formData.get('letter')?.toString() || undefined,
      easterEgg,
      sender: formData.get('sender')?.toString() || undefined,
    };

    const id = createBlessing(input);

    // Build full URL
    const host = request.headers.get('host') || 'localhost:3000';
    const protocol = request.headers.get('x-forwarded-proto') || 'http';
    const blessingUrl = `${protocol}://${host}/b/${id}`;

    // Generate QR code
    const qrcodeDataUrl = await QRCode.toDataURL(blessingUrl, {
      width: 400,
      margin: 2,
      color: { dark: '#d4a843', light: '#13131a' },
    });

    return NextResponse.json({
      success: true,
      id,
      url: blessingUrl,
      qrcode: qrcodeDataUrl,
    });
  } catch (err) {
    console.error('Create error:', err);
    return NextResponse.json(
      { success: false, error: '创建失败，请稍后重试' },
      { status: 500 }
    );
  }
}
```

```tsx
import { useState, useRef } from 'react';
import { PHOTO_ACCEPT, SITE_CONFIG } from '@/lib/config';

interface Props {
  onPhotosChange: (files: File[]) => void;
}

export function PhotoUploader({ onPhotosChange }: Props) {
  const [previews, setPreviews] = useState<string[]>([]);
  const inputRef = useRef<HTMLInputElement>(null);

  function handleFiles(files: FileList | null) {
    if (!files) return;
    const fileArr = Array.from(files).slice(0, SITE_CONFIG.maxPhotos);
    const urls = fileArr.map((f) => URL.createObjectURL(f));
    setPreviews((prev) => [...prev, ...urls].slice(0, SITE_CONFIG.maxPhotos));

    setPreviews((prev) => prev.filter((_, i) => i !== index));

        type="file"
        accept={PHOTO_ACCEPT}

        className="hidden"
```

```tsx
'use client';
import { useState, useRef } from 'react';

interface Props {
  onVoiceReady: (blob: Blob) => void;
  onClear: () => void;
}

export function VoiceRecorder({ onVoiceReady, onClear }: Props) {
  const [recording, setRecording] = useState(false);
  const [audioUrl, setAudioUrl] = useState<string | null>(null);
  const mediaRecorderRef = useRef<MediaRecorder | null>(null);
  const chunksRef = useRef<Blob[]>([]);
  const streamRef = useRef<MediaStream | null>(null);

  async function startRecording() {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      streamRef.current = stream;
      const recorder = new MediaRecorder(stream, { mimeType: 'audio/webm' });
      mediaRecorderRef.current = recorder;
      chunksRef.current = [];

      recorder.ondataavailable = (e) => {
        if (e.data.size > 0) chunksRef.current.push(e.data);
      };

      recorder.onstop = () => {
        const blob = new Blob(chunksRef.current, { type: 'audio/webm' });
        const url = URL.createObjectURL(blob);
        setAudioUrl(url);
        onVoiceReady(blob);
        stream.getTracks().forEach((t) => t.stop());
      };

      recorder.start();
      setRecording(true);
    } catch {
      alert('无法访问麦克风，请检查浏览器权限');
    }
  }

  function stopRecording() {
    mediaRecorderRef.current?.stop();
    setRecording(false);
  }

  function clearVoice() {
    setAudioUrl(null);
    onClear();
  }

  return (
    <div className="space-y-3">
      <label className="block text-sm font-medium text-text-secondary">语音祝福 (可选)</label>
      {audioUrl ? (
        <div className="space-y-2">
          <audio controls src={audioUrl} className="w-full h-10 rounded-lg" />
          <button
            type="button"
            onClick={clearVoice}
            className="text-xs text-text-secondary underline hover:text-accent-warm"
          >
            删除录音
          </button>
        </div>
      ) : (
        <button
          type="button"
          onClick={recording ? stopRecording : startRecording}
          className={`px-6 py-3 rounded-full text-sm font-medium transition-all ${
            recording
              ? 'bg-red-500/20 text-red-400 border border-red-500/40 animate-pulse'
              : 'bg-white/5 text-text-secondary border border-white/10 hover:bg-white/10'
          }`}
        >
          {recording ? '⏹ 停止录制' : '🎙 录制语音祝福'}
        </button>
      )}
    </div>
  );
}
```

```tsx
import { PhotoUploader } from './PhotoUploader';
import { VoiceRecorder } from './VoiceRecorder';
import { VideoUploader } from './VideoUploader';
import type { CreateApiResponse } from '@/lib/types';

interface Props {
  onSuccess: (data: CreateApiResponse) => void;
  onError: (msg: string) => void;
}

export function UploadForm({ onSuccess, onError }: Props) {
  const [submitting, setSubmitting] = useState(false);
  const photoFilesRef = useRef<File[]>([]);
  const voiceBlobRef = useRef<Blob | null>(null);
  const videoFileRef = useRef<File | null>(null);

  async function handleSubmit(e: FormEvent<HTMLFormElement>) {
    e.preventDefault();
    setSubmitting(true);

    const form = e.currentTarget;
    const formData = new FormData(form);

    // Append photos
    photoFilesRef.current.forEach((f) => formData.append('photos', f));
    // Append voice
    if (voiceBlobRef.current) {
      formData.append('voice', voiceBlobRef.current, 'blessing.webm');
    }
    // Append video
    if (videoFileRef.current) {
      formData.append('video', videoFileRef.current);
    }

    try {
      const res = await fetch('/api/create', { method: 'POST', body: formData });
      const data: CreateApiResponse = await res.json();
      if (!res.ok || !data.success) throw new Error(data.error || '创建失败');
      onSuccess(data);
    } catch (err: any) {
      onError(err.message || '网络错误');
    } finally {
      setSubmitting(false);
    }
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-6">
      {/* Name */}
      <div>
        <label htmlFor="name" className="block text-sm font-medium text-text-secondary mb-2">
          寿星姓名 <span className="text-accent-warm">*</span>
        </label>
        <input
          id="name"
          name="name"
          type="text"
          required
          maxLength={50}
          placeholder="输入寿星的姓名"
          className="w-full px-4 py-3 bg-bg-elevated border border-white/10 rounded-xl text-text-primary placeholder:text-text-muted focus:outline-none focus:border-accent-gold/50 transition-colors"
        />
      </div>

      {/* Birthday */}
      <div>
        <label htmlFor="birthday" className="block text-sm font-medium text-text-secondary mb-2">
          生日日期
        </label>
        <input
          id="birthday"
          name="birthday"
          type="date"
          className="w-full px-4 py-3 bg-bg-elevated border border-white/10 rounded-xl text-text-primary focus:outline-none focus:border-accent-gold/50 transition-colors"
        />
      </div>

      {/* Photos */}
      <PhotoUploader
        onPhotosChange={(files) => { photoFilesRef.current = files; }}
      />

      {/* Voice */}
      <VoiceRecorder
        onVoiceReady={(blob) => { voiceBlobRef.current = blob; }}
        onClear={() => { voiceBlobRef.current = null; }}
      />

      {/* Video */}
      <VideoUploader
        onVideoReady={(file) => { videoFileRef.current = file; }}
        onClear={() => { videoFileRef.current = null; }}
      />

      {/* Letter */}
      <div>
        <label htmlFor="letter" className="block text-sm font-medium text-text-secondary mb-2">
          生日信 (可选)
        </label>
        <textarea
          id="letter"
          name="letter"
          rows={4}
          maxLength={1000}
          placeholder="写下你想对寿星说的话..."
          className="w-full px-4 py-3 bg-bg-elevated border border-white/10 rounded-xl text-text-primary placeholder:text-text-muted focus:outline-none focus:border-accent-gold/50 transition-colors resize-none"
        />
      </div>

      {/* Sender */}
      <div>
        <label htmlFor="sender" className="block text-sm font-medium text-text-secondary mb-2">
          署名 (可选)
        </label>
        <input
          id="sender"
```

```tsx
// src/components/ui/ParticleField.tsx
'use client';
import { useEffect, useRef } from 'react';

interface Particle {
  x: number; y: number; vx: number; vy: number;
  size: number; opacity: number; life: number; maxLife: number;
}

interface Props {
  count?: number;
  className?: string;
  color?: string;
}

export function ParticleField({ count = 80, className = '', color = '255,255,255' }: Props) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    let animId: number;
    let particles: Particle[] = [];

    function resize() {
      canvas!.width = window.innerWidth;
      canvas!.height = window.innerHeight;
    }
    resize();
    window.addEventListener('resize', resize);

    function createParticle(x?: number, y?: number): Particle {
      return {
        x: x ?? Math.random() * canvas!.width,
        y: y ?? Math.random() * canvas!.height,
        vx: (Math.random() - 0.5) * 0.6,
        vy: (Math.random() - 0.5) * 0.6 - 0.3,
        size: Math.random() * 2.5 + 0.5,
        opacity: Math.random() * 0.6 + 0.1,
        life: 0,
        maxLife: Math.random() * 300 + 200,
      };
    }

    // Initial particles spread from center
    for (let i = 0; i < count; i++) {
      particles.push(createParticle(canvas.width / 2, canvas.height / 2));
    }

    function animate() {
      if (!ctx || !canvas) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      for (let i = particles.length - 1; i >= 0; i--) {
        const p = particles[i];
        p.x += p.vx;
        p.y += p.vy;
        p.life++;

        const fadeRatio = p.life < 30 ? p.life / 30 : 1 - (p.life - 30) / p.maxLife;
        const alpha = p.opacity * Math.max(0, fadeRatio);

        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(${color},${alpha})`;
        ctx.fill();

        if (p.life > p.maxLife) {
          particles[i] = createParticle();
        }
      }

      animId = requestAnimationFrame(animate);
    }

    animate();

    return () => {
      cancelAnimationFrame(animId);
      window.removeEventListener('resize', resize);
    };
  }, [count, color]);

  return <canvas ref={canvasRef} className={`fixed inset-0 pointer-events-none z-0 ${className}`} />;
}
```

```tsx
// src/components/ui/ParallaxLayer.tsx
'use client';
import { useEffect, useRef, ReactNode } from 'react';

interface Props {
  children: ReactNode;
  speed?: number; // 0 = static, 1 = normal scroll, 0.5 = half speed
  className?: string;
}

export function ParallaxLayer({ children, speed = 0.5, className = '' }: Props) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const el = ref.current;
    if (!el || typeof window === 'undefined') return;

    function onScroll() {
      const rect = el!.getBoundingClientRect();
      const scrolled = window.innerHeight - rect.top;
      const offset = scrolled * (1 - speed);
      el!.style.transform = `translateY(${offset * 0.3}px)`;
    }

    window.addEventListener('scroll', onScroll, { passive: true });
    onScroll();
    return () => window.removeEventListener('scroll', onScroll);
  }, [speed]);

  return (
    <div ref={ref} className={className}>
      {children}
    </div>
  );
}
```

```tsx
import { ReactNode } from 'react';

interface Props {
  children: ReactNode;
  className?: string;
  delay?: number;
  direction?: 'up' | 'down' | 'left' | 'right';
  duration?: number;
}

const directionMap = {
  up: { y: 40 },
  down: { y: -40 },
  left: { x: 40 },
  right: { x: -40 },
};

export function FadeReveal({ children, className = '', delay = 0, direction = 'up', duration = 0.8 }: Props) {
  const offset = directionMap[direction];

  return (
    <motion.div
      initial={{ opacity: 0, ...offset }}
      whileInView={{ opacity: 1, x: 0, y: 0 }}
      viewport={{ once: true, margin: '-80px' }}
      transition={{ duration, delay, ease: [0.25, 0.1, 0.25, 1] }}
      className={className}
    >
      {children}
    </motion.div>
  );
}
```

```tsx
'use client';
import { useEffect, useRef } from 'react';

interface Props {
  isPlaying: boolean;
  barCount?: number;
  className?: string;
}

export function AudioVisualizer({ isPlaying, barCount = 32, className = '' }: Props) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!isPlaying) return;
    const bars = containerRef.current?.children;
    if (!bars) return;

    let animId: number;
    function animate() {
      for (let i = 0; i < bars.length; i++) {
        const bar = bars[i] as HTMLElement;
        const h = 8 + Math.sin(Date.now() * 0.003 + i * 0.4) * 20 + Math.random() * 16;
        bar.style.height = `${h}px`;
      }
      animId = requestAnimationFrame(animate);
    }
    animate();
    return () => cancelAnimationFrame(animId);
  }, [isPlaying]);

  return (
    <div ref={containerRef} className={`flex items-end justify-center gap-[2px] ${className}`}>
      {Array.from({ length: barCount }).map((_, i) => (
        <div
          key={i}
          className="w-[3px] bg-accent-warm/70 rounded-full transition-all duration-75"
          style={{ height: isPlaying ? '10px' : '2px', opacity: isPlaying ? 0.8 : 0.3 }}
        />
      ))}
    </div>
  );
}
```

```tsx
import { useState, useRef, useEffect } from 'react';

  const audioRef = useRef<HTMLAudioElement>(null);
  const sectionRef = useRef<HTMLElement>(null);
```

```tsx
// src/components/film/PhotoTimeline.tsx
'use client';
import { useEffect, useRef } from 'react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { FadeReveal } from '@/components/ui/FadeReveal';

gsap.registerPlugin(ScrollTrigger);

interface Props {
  photos: string[];
}

export function PhotoTimeline({ photos }: Props) {
  const sectionRef = useRef<HTMLElement>(null);
  const stripRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const ctx = gsap.context(() => {
      const strip = stripRef.current;
      if (!strip || photos.length === 0) return;

      const totalWidth = strip.scrollWidth - window.innerWidth;

      gsap.to(strip, {
        x: -totalWidth,
        ease: 'none',
        scrollTrigger: {
          trigger: sectionRef.current,
          start: 'top top',
          end: `+=${totalWidth + window.innerHeight}`,
          scrub: 1,
          pin: sectionRef.current,
          anticipatePin: 1,
        },
      });
    }, sectionRef);

    return () => ctx.revert();
  }, [photos]);

  if (photos.length === 0) return null;

  return (
    <section ref={sectionRef} className="relative h-screen bg-bg-deep overflow-hidden">
      <FadeReveal className="absolute top-8 left-0 right-0 text-center z-10 pointer-events-none">
```

```tsx
      { threshold: 0.5 }

    return () => observer.disconnect();
```

```tsx
// src/components/film/FireworksFinale.tsx
'use client';
import { useEffect, useRef } from 'react';

interface Spark {
  x: number; y: number; vx: number; vy: number;
  color: string; alpha: number; decay: number; size: number;
}

export function FireworksFinale() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const sectionRef = useRef<HTMLElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    let animId: number;
    let sparks: Spark[] = [];
    const colors = ['#d4a843', '#e07040', '#60a0c0', '#f0f0f5', '#e0a060'];
    let lastLaunch = 0;

    function resize() {
      canvas!.width = window.innerWidth;
      canvas!.height = window.innerHeight;
    }
    resize();
    window.addEventListener('resize', resize);

    function launchFirework(x: number, y: number) {
      const count = 30 + Math.floor(Math.random() * 40);
      const color = colors[Math.floor(Math.random() * colors.length)];
      for (let i = 0; i < count; i++) {
        const angle = (Math.PI * 2 * i) / count + (Math.random() - 0.5) * 0.3;
        const speed = 1.5 + Math.random() * 3;
        sparks.push({
          x, y,
          vx: Math.cos(angle) * speed,
          vy: Math.sin(angle) * speed,
          color,
          alpha: 1,
          decay: 0.008 + Math.random() * 0.015,
          size: 1 + Math.random() * 2,
        });
      }
    }

    function animate(time: number) {
      if (!ctx || !canvas) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Launch new firework every 800-2000ms
      if (time - lastLaunch > 800 + Math.random() * 1200) {
        launchFirework(
          canvas.width * 0.2 + Math.random() * canvas.width * 0.6,
          canvas.height * 0.3 + Math.random() * canvas.height * 0.4,
        );
        lastLaunch = time;
      }

      // Update and draw sparks
      for (let i = sparks.length - 1; i >= 0; i--) {
        const s = sparks[i];
        s.vy += 0.02; // gravity
        s.x += s.vx;
        s.y += s.vy;
        s.alpha -= s.decay;

        if (s.alpha <= 0) {
          sparks.splice(i, 1);
          continue;
        }

        ctx.beginPath();
        ctx.arc(s.x, s.y, s.size, 0, Math.PI * 2);
        ctx.fillStyle = s.color.replace(')', `,${s.alpha})`).replace('rgb', 'rgba');
        if (s.color.startsWith('#')) {
          // Convert hex to rgba for alpha
          ctx.globalAlpha = s.alpha;
          ctx.fillStyle = s.color;
        }
        ctx.fill();
        ctx.globalAlpha = 1;
      }

      animId = requestAnimationFrame(animate);
    }

    // IntersectionObserver to detect when section is visible
    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          animId = requestAnimationFrame(animate);
        } else {
          cancelAnimationFrame(animId);
        }
      },
      { threshold: 0.6 }
    );
    observer.observe(sectionRef.current!);

    return () => {
      cancelAnimationFrame(animId);
      observer.disconnect();
      window.removeEventListener('resize', resize);
    };
  }, []);

  return (
    <section
      ref={sectionRef}
      className="relative h-screen flex flex-col items-center justify-center bg-bg-deep overflow-hidden"
    >
      <canvas ref={canvasRef} className="absolute inset-0" />

      <div className="relative z-10 text-center px-6">
        <p className="text-text-muted text-xs tracking-[0.3em] uppercase mb-4">For You</p>
        <h2 className="text-4xl md:text-5xl font-light text-accent-gold tracking-wider mb-4">
          生日快乐
        </h2>
        <p className="text-text-secondary text-base font-light">
          愿每一岁都奔走在热爱里
        </p>
      </div>
    </section>
  );
}
```
